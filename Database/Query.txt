28/6/2023 --- Table to Add api ---

CREATE DATABASE apimonitoring;

USE apimonitoring;

create table ADDAPI(
	API_ID INT NOT NULL  auto_increment ,
	API_NAME  varchar(50) NOT NULL,

	PRIMARY KEY (API_ID),
    UNIQUE(API_NAME)
);



30/6/2023 -- Table to add url -----

create table addapi_url(
	URL_ID INT NOT NULL  auto_increment,
	API_ID INT NOT NULL,
	API_URL  varchar(100) NOT NULL ,

	PRIMARY KEY (URL_ID),
    FOREIGN KEY (API_Id) REFERENCES addapi(API_Id)
);


30/6/2023 -- Table to add url Status  -----

create table ADDAPI_STATUS(
STATUS_ID INT NOT NULL  auto_increment,
URL_ID INT NOT NULL,
URL_STATUS TINYINT NOT NULL DEFAULT 1,
RESPONSE_TIME FLOAT,

 

PRIMARY KEY (STATUS_ID),
FOREIGN KEY (URL_ID) REFERENCES ADDAPI_URL(URL_ID)
);
